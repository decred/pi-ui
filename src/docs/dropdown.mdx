import { useState } from "react";
import { Playground, Props } from "docz";
import { Dropdown, Tabs, Tab } from "../index";

# Dropdown

## Properties
<Props of={Dropdown} />

## Usage

### Simple Dropdown

<Playground>
  {() => {
    const [activeTabIndex, setActiveTabIndex] = useState(0);
    return (
      <>
      	<Dropdown title='click me'>
	        <Tabs onSelectTab={setActiveTabIndex} activeTabIndex={activeTabIndex} vertical>
	            <Tab label="tab 1" />
	            <Tab label="tab 2" />
	            <Tab label="tab 3" />
	        </Tabs>
        </Dropdown>
      </>
    );
  }}
</Playground>

### Simple Dropdown with Count
<Playground>
