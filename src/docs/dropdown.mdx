import { useState } from "react";
import { Playground, Props } from "docz";
import { Dropdown, Tabs, Tab } from "../index";

# Dropdown

## Properties
<Props of={Dropdown} />

## Usage

### Simple Dropdown

<Playground>
  {() => {
    const [activeTabIndex, setActiveTabIndex] = useState(0);
    const [showDropdown, setShowDropdown] = useState(false)
    const openDropdown = () => setShowDropdown(true)
    const closeDropdown = () => setShowDropdown(false)
    return (
      <>
      	<Dropdown title='click me' show={showDropdown} handleClick={showDropdown ? closeDropdown : openDropdown}>
	        <li>My proposals</li>
	        <li>Account</li>
	        <li>Logout</li>
        </Dropdown>
      </>
    );
  }}
</Playground>

### Dropdown with Tabs

<Playground>
  {() => {
    const [activeTabIndex, setActiveTabIndex] = useState(0);
    const [showDropdown, setShowDropdown] = useState(false)
    const openDropdown = () => setShowDropdown(true)
    const closeDropdown = () => setShowDropdown(false)
    return (
      <>
      	<Dropdown title='click me' show={showDropdown} handleClick={showDropdown ? closeDropdown : openDropdown}>
	        <Tabs onSelectTab={setActiveTabIndex} activeTabIndex={activeTabIndex} vertical>
	            <Tab label="tab 1" />
	            <Tab label="tab 2" />
	            <Tab label="tab 3" />
	        </Tabs>
        </Dropdown>
      </>
    );
  }}
</Playground>
