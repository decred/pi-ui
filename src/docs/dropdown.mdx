import { useState } from "react";
import { Playground, Props } from "docz";
import { Dropdown, DropdownItem, Tabs, Tab, Button } from "../index";

# Dropdown

## Dropdown props
<Props of={Dropdown} />

## DropdownItem props
<Props of={DropdownItem} />


## Usage

### Uncontrolled dropdown

<Playground>
  {() => {
    return (
      <>
      	<Dropdown 
      		title='username'>
	        <DropdownItem  onClick={() => alert('my proposals')}>
            My proposals
          </DropdownItem>
          <DropdownItem  onClick={() => alert('account')}>
            Account
          </DropdownItem>
          <DropdownItem  onClick={() => alert('logout')}>
            Logout
          </DropdownItem>
        </Dropdown>
      </>
    );
  }}
</Playground>

### Controlled Dropdown

<Playground>
  {() => {
    const [showDropdown, setShowDropdown] = useState(false)
    const openDropdown = () => setShowDropdown(true)
    const closeDropdown = () => setShowDropdown(false)
    return (
      <>
      	<Dropdown 
      		title='username' 
      		show={showDropdown} 
      		onDropdownClick={showDropdown ? closeDropdown : openDropdown}>
	        <DropdownItem  onClick={() => alert('my proposals')}>
            My proposals
          </DropdownItem>
          <DropdownItem  onClick={() => alert('account')}>
            Account
          </DropdownItem>
          <DropdownItem  onClick={() => alert('logout')}>
            Logout
          </DropdownItem>
        </Dropdown>
      </>
    );
  }}
</Playground>

### Dropdown with Alternative Dropdown Trigger

<Playground>
  {() => {

    const CustomTrigger = ({ onClick, open }) => <Button onClick={onClick}>{open ? "close" : "open"}</Button>
    return (
      <>
      	<Dropdown 
      		customDropdownTrigger={CustomTrigger}
      		closeOnOutsideClick={true}>
	        <DropdownItem  onClick={() => alert('my proposals')}>
            My proposals
          </DropdownItem>
          <DropdownItem  onClick={() => alert('account')}>
            Account
          </DropdownItem>
          <DropdownItem  onClick={() => alert('logout')}>
            Logout
          </DropdownItem>
        </Dropdown>
      </>
    );
  }}
</Playground>
