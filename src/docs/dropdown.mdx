import { useState } from "react";
import { Playground, Props } from "docz";
import { Dropdown, DropdownItem, Tabs, Tab, Button } from "../index";

# Dropdown

## Dropdown props
<Props of={Dropdown} />

## DropdownItem props
<Props of={DropdownItem} />


## Usage

### Uncontrolled dropdown

<Playground>
  {() => {
    return (
      <>
      	<Dropdown 
      		title='mariamaria'>
	        <DropdownItem label='My proposals' onClick={() => alert('my proposals')}/>
	        <DropdownItem label='Account' onClick={() => alert('account')}/>
	        <DropdownItem label='Logout' onClick={() => alert('my props')}/>
        </Dropdown>
      </>
    );
  }}
</Playground>

### Controlled Dropdown

<Playground>
  {() => {
    const [showDropdown, setShowDropdown] = useState(false)
    const openDropdown = () => setShowDropdown(true)
    const closeDropdown = () => setShowDropdown(false)
    return (
      <>
      	<Dropdown 
      		title='mariamaria' 
      		show={showDropdown} 
      		onDropdownClick={showDropdown ? closeDropdown : openDropdown}>
	        <DropdownItem label='My proposals' handleItemClick={() => alert('my proposals')}/>
	        <DropdownItem label='Account' handleItemClick={() => alert('account')}/>
	        <DropdownItem label='Logout' handleItemClick={() => alert('my props')}/>
        </Dropdown>
      </>
    );
  }}
</Playground>

### Dropdown with Alternative Dropdown Trigger

<Playground>
  {() => {

    const CustomTrigger = ({ onClick, open }) => <Button onClick={onClick}>{open ? "close" : "open"}</Button>
    return (
      <>
      	<Dropdown 
      		customDropdownTrigger={CustomTrigger}
      		closeOnOutsideClick={true}>
	        <DropdownItem label='My proposals' handleItemClick={() => alert('my proposals')}/>
	        <DropdownItem label='Account' handleItemClick={() => alert('account')}/>
	        <DropdownItem label='Logout' handleItemClick={() => alert('my props')}/>
        </Dropdown>
      </>
    );
  }}
</Playground>
