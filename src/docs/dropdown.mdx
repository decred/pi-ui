## Usage

### Simple Dropdown

<Playground>
  {() => {
    const [activeTabIndex, setActiveTabIndex] = useState(0);
    const [showDropdown, setShowDropdown] = useState(false)
    const openDropdown = () => setShowDropdown(true)
    const closeDropdown = () => setShowDropdown(false)
    return (
      <>
      	<Dropdown 
      		title='mariamaria' 
      		show={showDropdown} 
      		onDropdownClick={showDropdown ? closeDropdown : openDropdown}
      		closeOnOutsideClick={true}>
	        <DropdownItem label='My proposals' handleItemClick={() => alert('my proposals')}/>
	        <DropdownItem label='Account' handleItemClick={() => alert('account')}/>
	        <DropdownItem label='Logout' handleItemClick={() => alert('my props')}/>
        </Dropdown>
      </>
    );
  }}
</Playground>

### Dropdown with Tabs
