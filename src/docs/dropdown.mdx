import { useState } from "react";
import { Playground, Props } from "docz";
import { Dropdown, DropdownItem, Tabs, Tab } from "../index";

# Dropdown

## Properties
<Props of={Dropdown} />

## Usage

### Simple Dropdown

<Playground>
  {() => {
    const [activeTabIndex, setActiveTabIndex] = useState(0);
    const [showDropdown, setShowDropdown] = useState(false)
    const openDropdown = () => setShowDropdown(true)
    const closeDropdown = () => setShowDropdown(false)
    return (
      <>
      	<Dropdown title='mariamaria' show={showDropdown} handleClick={showDropdown ? closeDropdown : openDropdown}>
	        <DropdownItem label='My proposals'/>
	        <DropdownItem label='Account'/>
	        <DropdownItem label='Logout'/>
        </Dropdown>
      </>
    );
  }}
</Playground>

### Dropdown with Tabs

<Playground>
