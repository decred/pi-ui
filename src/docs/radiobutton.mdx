import { useState, Fragment } from "react";
import { Playground, Props } from "docz";
import { RadioButtonGroup } from "../index";
import styles from "../css/base.css";

# Radio Button Group

## Properties
<Props of={RadioButtonGroup} />

## Usage

<Playground>
  {() => {
    const options = [{ label: "foo", value: 1, description: "this is foo!"}, 
      {label: "bar", value: 2, description: "this is bar!"}];
    const [value, setValue] = useState(1)
    const handleChange = (option) => {
        setValue(option.value);
    }
    return (
      <Fragment>
      <RadioButtonGroup vertical  label="Select an option" options={options} value={value} onChange={handleChange} />
      </Fragment>
    );
  }    
}
</Playground>
