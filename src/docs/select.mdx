### Basic

<Playground>
  {() => {
    const options = [{
      value: 'top',
      label: 'Top'
    }, {
      value: 'new',
      label: 'New'
    }, {
      value: 'old',
      label: 'Old'
    }];
    const [value, onChange] = useState({
      value: 'new',
      label: 'New'
    });
    const [inputValue, onInputChange] = useState("");
    const [disabled, setDisabled] = useState(false);
    const [clearable, setClearable] = useState(false);
    const [separator, setSeparator] = useState(false);
    const [searchable, setSearchable] = useState(true);
    const [filter, setFilter] = useState("");

    return (
      <Fragment>
      	<Select 
          disabled={disabled}
          clearable={clearable}
          separator={separator}
          searchable={searchable}
          options={options}
          value={value}
          onChange={onChange}
          inputValue={inputValue}
          onInputChange={onInputChange}
          optionsFilter={({label}) => (label !== filter)}
          label={"Basic select"}
        />
        <br/>
        <Checkbox id="basic-select-disabled" label={"Disabled"} checked={disabled} onChange={() => setDisabled(!disabled)}/>
        <br/>
        <Checkbox id="basic-select-clearable" label={"Clearable"} checked={clearable} onChange={() => setClearable(!clearable)}/>
        <br/>
        <Checkbox id="basic-select-separator" label={"Separator"} checked={separator} onChange={() => setSeparator(!separator)}/>
        <br/>
        <Checkbox id="basic-select-searchable" label={"Searchable"} checked={searchable} onChange={() => setSearchable(!searchable)}/>
        <br/>
        <Text>Don't show option:</Text><TextInput id="basic-select-filter" label={""} onChange={(e) => setFilter(e.target.value)}/>
      </Fragment>
    );
  }}
</Playground>

### Custom renderers

<Playground>
  {() => {
    const options = [{
      value: 'top',
      label: 'Top'
    }, {
      value: 'new',
      label: 'New'
    }, {
      value: 'old',
      label: 'Old'
    }];
    const [value, onChange] = useState({
      value: 'new',
      label: 'New'
    });
    const customRenderer = ({label}) => (
      <div style={{display: "flex", alignItems: "center"}}>
        {
          label.split('').reduce((map, labelLetter) => [
            ...map,
            <h3>{labelLetter}</h3>, 
            <div
              style={{
                width: "2rem",
                height: "2rem",
                margin: "0 1rem",
                borderRadius: "100%",
                backgroundColor: "#" + ((1 << 24) * Math.random() | 0).toString(16)
              }}
            />
          ], [])
        }
      </div>
    );

    return (
      <Fragment>
      	<Select 
          options={options}
          value={value}
          onChange={onChange}
          optionRenderer={customRenderer}
          valueRenderer={customRenderer}
          label={"Custom renderers"}
        />
      </Fragment>
    );
  }}
</Playground>

# Async select

## Async select props
<Props of={AsyncSelect} />

## Usage

<Playground>
  {() => {
    const options = [{
      value: 'top',
      label: 'Top'
    }, {
      value: 'new',
      label: 'New'
    }, {
      value: 'old',
      label: 'Old'
    }];
    const [value, onChange] = useState({
      value: 'new',
      label: 'New'
    });
    const [inputValue, onInputChange] = useState("");
    const [disabled, setDisabled] = useState(false);
    const [clearable, setClearable] = useState(false);
    const [separator, setSeparator] = useState(false);

    const colors = [{
      value: 'yellow',
      label: 'Yellow'
    }, {
      value: 'orange',
      label: 'Orange'
    }, {
      value: 'red',
      label: 'Red'
    }, {
      value: 'purple',
      label: 'Purple'
    }, {
      value: 'blue',
      label: 'Blue'
    }, {
      value: 'green',
      label: 'Green'
    }];

    const optionsFilter = (_inputValue) =>  colors.filter(color =>
      color.label.toLowerCase().includes(_inputValue.toLowerCase())
    );

    const promiseOptions = _inputValue =>
      new Promise(resolve => {
        setTimeout(() => {
          resolve(optionsFilter(_inputValue));
        }, 1000);
      });

    const defaultOptions = [{
      label: "Bottom",
      value: "bottom"
    }];

    return (
      <Fragment>
      	<AsyncSelect 
          disabled={disabled}
          clearable={clearable}
          separator={separator}
          options={options}
          value={value}
          onChange={onChange}
          inputValue={inputValue}
          onInputChange={onInputChange}
          label={"Async select"}
          loadOptions={promiseOptions}
        />
        <br/>
        <Checkbox id="async-select-disabled" label={"Disabled"} checked={disabled} onChange={() => setDisabled(!disabled)}/>
        <br/>
        <Checkbox id="async-select-clearable" label={"Clearable"} checked={clearable} onChange={() => setClearable(!clearable)}/>
        <br/>
        <Checkbox id="async-select-separator" label={"Separator"} checked={separator} onChange={() => setSeparator(!separator)}/>
      </Fragment>
    );
  }}
</Playground>
