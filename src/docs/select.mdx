import { useState } from "react";
import { Playground, Props } from "docz";
import { Select } from "../index";

# Select

## Select props
<Props of={Select} />

## Usage

### Basic select

<Playground>
  {() => {
    const options = [{
      value: 'top',
      label: 'Top'
    }, {
      value: 'new',
      label: 'New'
    }, {
      value: 'old',
      label: 'Old'
    }];
    const [selected, setSelected] = useState({
      value: 'old',
      label: 'Old'
    });
    const selectChangeHandler = (option) => {
      setSelected(option);
    };
    return (
      <>
      	<Select 
          options={options}
          value={selected}
          onChange={selectChangeHandler}
        />
      </>
    );
  }}
</Playground>
