import { useState, Fragment } from "react";
import { Playground, Props } from "docz";
import { Paginator } from "../index";

# Paginator

## Properties

<Props of={Paginator} />

## Usage


### Default Paginator

<Playground>
  {() => {
    const [selectedPage, setSelectedPage] = useState(0);
    const onPageChange = ({ selected }) =>  setSelectedPage(selected);
    const pageCount = 20;
    return (
      <Fragment>
        <div>{`${selectedPage + 1}/${pageCount}`}</div>
        <Paginator pageCount={pageCount} onPageChange={onPageChange}/>
      </Fragment>
    );
  }}
</Playground>


### Paginator with custom paginationGap and marginPagesDisplayed

<Playground>
  {() => {
    const pageCount = 20;
    return (
      <Fragment>
        <Paginator pageCount={pageCount} paginationGap={2} marginPagesDisplayed={2}/>
      </Fragment>
    );
  }}
</Playground>
