## Properties

## Usage

### Simple Modal
<Playground>
  {() => {
    const [showModal, setShowModal] = useState(false);
    const openModal = () => setShowModal(true);
    const closeModal = () => setShowModal(false);
    return (
      <>
        <Button onClick={openModal}>Open Modal</Button>
        <Modal show={showModal} onClose={closeModal}>
          Simple Modal
        </Modal>
      </>
    );
  }}
</Playground>
