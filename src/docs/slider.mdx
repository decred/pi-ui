import { useState, Fragment } from "react";
import { Playground, Props } from "docz";
import { Slider, NumberInput, Checkbox } from "../index";

# Slider

## Slider props

<Props of={Slider} />

## Usage

### Single horizontal slider

<Playground>
  {() => {
    const [coordinates, setCoordinates] = useState({ handle1: 0, handle2: 40 });
    const [min, setMin] = useState(0);
    const [max, setMax] = useState(100);
    const [disabled, setDisabled] = useState(false);
    return (
      <Fragment>
        <Slider
          double={false}
          disabled={disabled}
          axis={"x"}
          max={max}
          min={min}
          handles={[
            {
              value: coordinates.handle1,
              onChange: (value) =>
                setCoordinates((coordinates) => ({
                  ...coordinates,
                  handle1: value
                }))
            }
          ]}
        />
        <br />
        <br />
        <Checkbox
          label="Disabled"
          id="single-horizontal-slider-disabled"
          checked={disabled}
          onChange={() => setDisabled(!disabled)}
        />
        <NumberInput
          label="Minimum"
          id="single-horizontal-slider-min"
          value={min}
          onChange={(e) => {
            const newLimit = parseInt(e.target.value);
            if (newLimit < max) setMin(newLimit);
          }}
        />
        <br />
        <NumberInput
          label="Maximum"
          id="single-horizontal-slider-max"
          value={max}
          onChange={(e) => {
            const newLimit = parseInt(e.target.value);
            if (newLimit > min) setMax(newLimit);
          }}
        />
      </Fragment>
    );
  }}
</Playground>
