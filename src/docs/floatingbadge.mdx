## Usage

<Playground>
	{() => {
    const [showBadge, setShowBadge] = useState(false);
    const openBadge = () => setShowBadge(true);
    const closeBadge = () => setShowBadge(false);
    return (
      <>
        <Button onClick={openBadge}>Open Badge</Button>

      </>
