## Usage

### Simple Tabs

<Playground>
  {() => {
    const [activeTabIndex, setActiveTabIndex] = useState(0);
    return (
      <>
        <Tabs onSelectTab={setActiveTabIndex} activeTabIndex={activeTabIndex}>
            <Tab label="tab 1" />
            <Tab label="tab 2" />
            <Tab label="tab 3" />
        </Tabs>
      </>
    );
  }}
</Playground>
